# ladybird

## Install

```bash
npm install --save ladybird
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'ladybird'
import 'ladybird/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
